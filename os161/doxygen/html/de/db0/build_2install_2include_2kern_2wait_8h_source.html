30#ifndef _KERN_WAIT_H_
31#define _KERN_WAIT_H_
32
33/*
34 * Definitions for wait().
35 */
36
37
38/* Flags for waitpid() and equivalent. */
39#define WNOHANG 1 /* Nonblocking. */
40#define WUNTRACED 2 /* Report stopping as well as exiting processes. */
41
42/* Special "pids" to wait for. */
43#define WAIT_ANY (-1) /* Any child process. */
44#define WAIT_MYPGRP 0 /* Any process in the same process group. */
45
46/*
47 * Result encoding.
48 *
49 * The lowest two bits say what happened; the rest encodes up to 30
50 * bits of exit code. Note that the traditional Unix encoding, which
51 * is different, wastes most of the bits and can only transmit 8 bits
52 * of exit code...
53 */
54#define _WWHAT(x) ((x)&3) /* lower two bits say what happened */
55#define _WVAL(x) ((x)>>2) /* the rest is the value */
56#define _MKWVAL(x) ((x)<<2) /* encode a value */
57
58/* Four things can happen... */
59#define __WEXITED 0 /* Process exited by calling _exit(). */
60#define __WSIGNALED 1 /* Process received a fatal signal. */
61#define __WCORED 2 /* Process dumped core on a fatal signal. */
62#define __WSTOPPED 3 /* Process stopped (and didn't exit). */
63
64/* Test macros, used by applications. */
65#define WIFEXITED(x) (_WWHAT(x)==__WEXITED)
66#define WIFSIGNALED(x) (_WWHAT(x)==__WSIGNALED || _WWHAT(x)==__WCORED)
67#define WIFSTOPPED(x) (_WWHAT(x)==__WSTOPPED)
68#define WEXITSTATUS(x) (_WVAL(x))
69#define WTERMSIG(x) (_WVAL(x))
70#define WSTOPSIG(x) (_WVAL(x))
71#define WCOREDUMP(x) (_WWHAT(x)==__WCORED)
72
73/* Encoding macros, used by the kernel to generate the wait result. */
74#define _MKWAIT_EXIT(x) (_MKWVAL(x)|__WEXITED)
75#define _MKWAIT_SIG(x) (_MKWVAL(x)|__WSIGNALED)
76#define _MKWAIT_CORE(x) (_MKWVAL(x)|__WCORED)
77#define _MKWAIT_STOP(x) (_MKWVAL(x)|__WSTOPPED)
78
79#endif /* _KERN_WAIT_H_ */
