38/*
39 * Time with fractional seconds. Important. Unfortunately, to be
40 * compatible, we need both timeval and timespec.
41 */
42
43struct timeval {
44 __time_t tv_sec; /* seconds */
45 __i32 tv_usec; /* microseconds */
46};
47
48struct timespec {
49 __time_t tv_sec; /* seconds */
50 __i32 tv_nsec; /* nanoseconds */
51};

58/* codes for the various timers */
59#define ITIMER_REAL 0 /* Real (wall-clock) time. */
60#define ITIMER_VIRTUAL 1 /* Virtual (when process is executing) time. */

63/* structure for setitimer/getitimer */
64struct itimerval {
65 struct timeval it_interval; /* Time to reload after expiry. */
