48/*
49 * Base array type (resizeable array of void pointers) and operations.
50 *
51 * create - allocate an array.
52 * destroy - destroy an allocated array.
53 * init - initialize an array in space externally allocated.
54 * cleanup - clean up an array in space externally allocated.
55 * num - return number of elements in array.
56 * get - return element no. INDEX.
57 * set - set element no. INDEX to VAL.
58 * preallocate - allocate space without changing size; may fail and
59 * return error.
60 * setsize - change size to NUM elements; may fail and return error.
61 * add - append VAL to end of array; return its index in INDEX_RET if
62 * INDEX_RET isn't null; may fail and return error.
63 * remove - excise entry INDEX and slide following entries down to
64 * close the resulting gap.
65 *
66 * Note that expanding an array with setsize doesn't initialize the new
67 * elements. (Usually the caller is about to store into them anyway.)
68 */
69
70struct array {
71 void **v;
72 unsigned num, max;
73};

88 * Inlining for base operations
89 */
90
91ARRAYINLINE unsigned
92array_num(const struct array *a)
93{
94 return a->num;
95}
96
97ARRAYINLINE void *
98array_get(const struct array *a, unsigned index)
99{
100 ARRAYASSERT(index < a->num);
101 return a->v[index];
102}
103
104ARRAYINLINE void
105array_set(const struct array *a, unsigned index, void *val)
106{
107 ARRAYASSERT(index < a->num);
108 a->v[index] = val;
109}
110
111ARRAYINLINE int
112array_add(struct array *a, void *val, unsigned *index_ret)
113{
114 unsigned index;
115 int ret;
116
117 index = a->num;
118 ret = array_setsize(a, index+1);
119 if (ret) {
120 return ret;
121 }
122 a->v[index] = val;
123 if (index_ret != NULL) {
124 *index_ret = index;
125 }
126 return 0;
127}

129/*
130 * Bits for declaring and defining typed arrays.
131 *
132 * Usage:
133 *
134 * DECLARRAY_BYTYPE(foo, bar) declares "struct foo", which is
135 * an array of pointers to "bar", plus the operations on it.
136 *
137 * DECLARRAY(foo) is equivalent to DECLARRAY_BYTYPE(fooarray, struct foo).
138 *
139 * DEFARRAY_BYTYPE and DEFARRAY are the same as DECLARRAY except that
140 * they define the operations, and both take an extra argument INLINE.
141 * For C99 this should be INLINE in header files and empty in the
142 * master source file, the same as the usage of ARRAYINLINE above and
143 * in array.c.
144 *
145 * Example usage in e.g. item.h of some game:
146 *
147 * DECLARRAY_BYTYPE(stringarray, char);
148 * DECLARRAY(potion);
149 * DECLARRAY(sword);
150 *
151 * #ifndef ITEMINLINE
152 * #define ITEMINLINE INLINE
153 * #endif
154 *
155 * DEFARRAY_BYTYPE(stringarray, char, ITEMINLINE);
156 * DEFARRAY(potion, ITEMINLINE);
157 * DEFARRAY(sword, ITEMINLINE);
158 *
159 * Then item.c would do "#define ITEMINLINE" before including item.h.
160 *
161 * This creates types "struct stringarray", "struct potionarray",
162 * and "struct swordarray", with operations such as "swordarray_num".
163 *
164 * The operations on typed arrays are the same as the operations on
165 * the base array, except typed.
166 */
167
