55static
56void
57fill(char *buf, size_t buflen)
58{
59 static const char *const names[22] = {
60 "Alhammitt", "Biffa", "Cennoreth", "Dastgandlen", "Enblith",
61 "Fenna", "Gull", "Hern", "Hildrida", "Kankredin", "Kialan",
62 "Lenina", "Manaliabrid", "Mayelbridwen", "Noreth", "Osfameron",
63 "Robin", "Tanamil", "Tanamoril", "Tanaqui", "Ynen", "Ynynen"
64 };
65
66 const char *name;
67 size_t len;
68
69 while (buflen > 4) {
70 name = names[random()%22];
71 len = strlen(name);
72 if (len < buflen) {
73 strcpy(buf, name);
74 buf += len;
75 buflen -= len;
76 if (buflen > 1) {
77 *buf = ' ';
78 buf++;
79 buflen--;
80 }
81 }
82 }
83 while (buflen > 1) {
84 *buf = '.';
85 buf++;
86 buflen--;
87 }
88 *buf = 0;
89}

104static
105void
106try(const char *first, ...)
107{
108 const char *args[20];
109 const char *s;
110 va_list ap;
111 int num;
112
113 assert(first != NULL);
114 args[0] = _PATH_MYSELF;
115 args[1] = first;
116 num = 2;
117
118 va_start(ap, first);
119 while (1) {
120 s = va_arg(ap, const char *);
121 if (s == NULL) {
122 break;
123 }
124 assert(num < 20);
125 args[num++] = s;
126 }
127 assert(num < 20);
128 args[num] = NULL;
129 execv(_PATH_MYSELF, (char **)args);
130 err(1, "execv");
131}

149static
150int
151check(int argc, char *argv[], const char *first, ...)
152{
153 const char *s;
154 va_list ap;
155 int pos;
156
157 pos = 1;
158 va_start(ap, first);
159 s = first;
160 while (s != NULL) {
161 if (pos == argc) {
162 /* not enough args */
163 return 0;
164 }
165 assert(pos < argc);
166 if (argv[pos] == NULL) {
167 /* woops */
168 warnx("argv[%d] is null but argc is %d", pos, argc);
169 return 0;
170 }
171 if (strcmp(argv[pos], s) != 0) {
172 /* does not match */
173 return 0;
174 }
175 s = va_arg(ap, const char *);
176 pos++;
177 }
178 if (pos != argc) {
179 /* too many args */
180 return 0;
181 }
182 if (argv[pos] != NULL) {
183 warnx("argv[argc] is not null");
184 return 0;
185 }
186 return 1;
187}

191checkmany(int argc, char *argv[], int num, const char *word)
192{
193 int i;
194
195 if (argc != num + 1) {
196 /* wrong number of args */
197 return 0;

199 for (i=1; i<argc; i++) {

201 /* doesn't match */
202 return 0;
203 }
204 }

211static
212void
213dumpargs(int argc, char *argv[])
214{
215 const char *s;
216 int i;
217
218 warnx("%d args", argc);
219 warnx("argv[0]: %s", argv[0]);
220 for (i=1; i<=argc; i++) {
221 s = argv[i];
222 if (s == NULL) {
223 warnx("argv[%d]: is null", i);
224 }
225 else if (!strcmp(s, word8)) {
226 warnx("argv[%d] is word8", i);
227 }
228 else if (!strcmp(s, word4050)) {
229 warnx("argv[%d] is word4050", i);
230 }
231 else if (!strcmp(s, word16320)) {
232 warnx("argv[%d] is word16320", i);
233 }
234 else if (!strcmp(s, word65500)) {
235 warnx("argv[%d] is word65500", i);
236 }
237 else if (strlen(s) < 72) {
238 warnx("argv[%d]: %s", i, s);
239 }
240 else {
241 warnx("argv[%d] is %zu bytes, begins %.64s",
242 i, strlen(s), s);
243 }
244 }

247int
248main(int argc, char *argv[])
249{
250 if (argc < 0) {
251 err(1, "argc is negative!?");
252 }
253
254 prepwords();
255 assert(strlen(word8) == 8);
256 assert(strlen(word4050) == 4050);
257 assert(strlen(word16320) == 16320);
258 assert(strlen(word65500) == 65500);
259
260 assert(ARG_MAX >= 65536);
261
262 if (argv == NULL || argc == 0 || argc == 1) {
263 /* no args -- start the test */
264 warnx("Starting.");
265
266 /*
267 * 1. Should always fit no matter what.
268 */
269 warnx("1. Execing with one 8-letter word.");
270 try(word8, NULL);
271 }
272 else if (check(argc, argv, word8, NULL)) {
273 /*
274 * 2. Fits in one page.
275 */
276 warnx("2. Execing with one 4050-letter word.");
277 try(word4050, NULL);
278 }
279 else if (check(argc, argv, word4050, NULL)) {
280 /*
281 * 3. Requires two pages but each word fits on a page.
282 */
283 warnx("3. Execing with two 4050-letter words.");
284 try(word4050, word4050, NULL);
285 }
286 else if (check(argc, argv, word4050, word4050, NULL)) {
287 /*
288 * 4. Requires the full 64K argv buffer, in large
289 * chunks, with a little space for slop. Each word
290 * fits on a page though. With null terminators and
291 * 4-byte pointers the size is 4085*16 = 65360, and
292 * with 8-byte pointers it would become 65424.
293 *
294 * Don't forget that argv[0] will be another 21 or 25
295 * bytes and some implementations may reasonably need
296 * to stash an ending NULL in the buffer too.
297 */
298 warnx("4. Execing with 16 4050-letter words.");
299 try(word4050, word4050, word4050, word4050,
300 word4050, word4050, word4050, word4050,
301 word4050, word4050, word4050, word4050,
302 word4050, word4050, word4050, word4050,
303 NULL);
304 }
305 else if (check(argc, argv,
306 word4050, word4050, word4050, word4050,
307 word4050, word4050, word4050, word4050,
308 word4050, word4050, word4050, word4050,
309 word4050, word4050, word4050, word4050,
310 NULL)) {
311 /*
312 * 5. Requires more than one page for a single word.
313 */
314 warnx("5. Execing with one 16320-letter word.");
315 try(word16320, NULL);
316 }
317 else if (check(argc, argv, word16320, NULL)) {
318 /*
319 * 6. Ditto but makes sure it works with two of them.
320 */
321 warnx("6. Execing with two 16320-letter words.");
322 try(word16320, word16320, NULL);
323 }
324 else if (check(argc, argv, word16320, word16320, NULL)) {
325 /*
326 * 7. Requires the full 64K argv buffer.
327 */
328 warnx("7. Execing with four 16320-letter words.");
329 try(word16320, word16320, word16320, word16320,
330 NULL);
331 }
332 else if (check(argc, argv, word16320, word16320,
333 word16320, word16320, NULL)) {
334 /*
335 * 8. Also requires the full 64K argv buffer, but with
336 * only one huge word.
337 */
338 warnx("8. Execing with one 65500-letter word.");
339 try(word65500, NULL);
340 }
341 else if (check(argc, argv, word65500, NULL)) {
342 /*
343 * 9. This fits on one page. Given 4-byte pointers,
344 * (8+1+4)*300 = 3900. With 8-byte pointers, it
345 * doesn't, but we aren't doing that. (Update this if
346 * we ever move to a 64-bit platform.)
347 */
348 assert((8+1+sizeof(char *))*300 < 4096);
349 warnx("9. Execing with 300 8-letter words.");
350 trymany(300, word8);
351 }
352 else if (checkmany(argc, argv, 300, word8)) {
353#if 1 /* enforce the full size */
354 /*
355 * 10. This requires the full 64K argv buffer.
356 * With 4-byte pointers, (8+1+4)*5020 = 65260.
357 * It also doesn't fit with 8-byte pointers.
358 *
359 * XXX for the time being, we'll allow less efficient
360 * implementations that use two pointers per word.
361 * Hence, (8+1+4+4)*3850 = 65450.
362 */
363 assert((8+1+sizeof(char *))*5020 < 65536);
364 assert((8+1+2*sizeof(char *))*3850 < 65536);
365 warnx("10. Execing with 3850 8-letter words.");
366 trymany(3850, word8);
367 }
368 else if (checkmany(argc, argv, 3850, word8)) {
369#else
370 /*
371 * 10a. This requires more than one page using small
372 * words. With 4-byte pointers, (8+1+4)*1000 = 13000.
373 */
374 warnx("10. Execing with 1000 8-letter words.");
375 trymany(1000, word8);
376 }
377 else if (checkmany(argc, argv, 1000, word8)) {
378#endif
379 warnx("Complete.");
380 return 0;
381 }
382 else {
383 warnx("Received unknown/unexpected args:");
384 dumpargs(argc, argv);
385 return 1;
386 }
387}
