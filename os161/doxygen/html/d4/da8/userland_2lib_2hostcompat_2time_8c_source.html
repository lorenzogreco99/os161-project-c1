40time_t
41__time(time_t *secs, unsigned long *nsecs)
42{
43 struct timeval tv;
44 if (gettimeofday(&tv, NULL) < 0) {
45 return -1;
46 }
47 if (secs) {
48 *secs = tv.tv_sec;
49 }
50 if (nsecs) {
51 *nsecs = tv.tv_usec * 1000;
52 }
53 return tv.tv_sec;
54}
