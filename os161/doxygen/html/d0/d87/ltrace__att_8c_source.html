36/* Lowest revision we support */
37#define LOW_VERSION 1
38/* Revision that supports ltrace_stop() */
39#define STOP_VERSION 2
40/* Revision that supports ltrace_prof() */
41#define PROF_VERSION 3
42
43struct ltrace_softc *
44attach_ltrace_to_lamebus(int ltraceno, struct lamebus_softc *sc)
45{
46 struct ltrace_softc *lt;
47 uint32_t drl;
48 int slot = lamebus_probe(sc, LB_VENDOR_CS161, LBCS161_TRACE,
49 LOW_VERSION, &drl);
50 if (slot < 0) {
51 return NULL;
52 }
53
54 lt = kmalloc(sizeof(struct ltrace_softc));
55 if (lt==NULL) {
56 return NULL;
57 }
58
59 (void)ltraceno; // unused
60
61 lt->lt_busdata = sc;
62 lt->lt_buspos = slot;
63 lt->lt_canstop = drl >= STOP_VERSION;
64 lt->lt_canprof = drl >= PROF_VERSION;
65
66 lamebus_mark(sc, slot);
67
68 return lt;
69}
