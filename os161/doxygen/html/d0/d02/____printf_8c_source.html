56/*
57 * Do we want to support "long long" types with %lld?
58 *
59 * Using 64-bit types with gcc causes gcc to emit calls to functions
60 * like __moddi3 and __divdi3. These need to be provided at link time,
61 * which can be a hassle; this switch is provided to help avoid
62 * needing them.
63 */
64#define USE_LONGLONG
65
66/*
67 * Define a type that holds the longest signed integer we intend to support.
68 */
69#ifdef USE_LONGLONG
70#define INTTYPE long long
71#else
72#define INTTYPE long
73#endif

76/*
77 * Space for a long long in base 8, plus a NUL, plus one
78 * character extra for slop.
79 *
80 * CHAR_BIT is the number of bits in a char; thus sizeof(long long)*CHAR_BIT
81 * is the number of bits in a long long. Each octal digit prints 3 bits.
82 * Values printed in larger bases will be shorter strings.
83 */
84#define NUMBER_BUF_SIZE ((sizeof(INTTYPE) * CHAR_BIT) / 3 + 2)

86/*
87 * Structure holding the state for printf.
88 */
89typedef struct {
90 /* Callback for sending printed string data */
91 void (*sendfunc)(void *clientdata, const char *str, size_t len);
92 void *clientdata;
93
94 /* The varargs argument pointer */
95 va_list ap;
96
97 /* Total count of characters printed */
98 int charcount;
99
100 /* Flag that's true if we are currently looking in a %-format */
101 int in_pct;
102
103 /* Size of the integer argument to retrieve */
104 enum {
105 INTSZ,
106 LONGSZ,
107#ifdef USE_LONGLONG
108 LLONGSZ,
109#endif
110 SIZETSZ,
111 } size;
112
113 /* The value of the integer argument retrieved */
114 unsigned INTTYPE num;
115
116 /* Sign of the integer argument (0 = positive; -1 = negative) */
117 int sign;
118
119 /* Field width (number of spaces) */
120 int spacing;
121
122 /* Flag: align to left in field instead of right */
123 int rightspc;
124
125 /* Character to pad to field size with (space or 0) */
126 int fillchar;
127
128 /* Number base to print the integer argument in (8, 10, 16) */
129 int base;
130
131 /* Flag: if set, print 0x before hex and 0 before octal numbers */
132 int baseprefix;
133
134 /* Flag: alternative output format selected with %#... */
135 int altformat;
136} PF;

138/*
139 * Send some text onward to the output.
140 *
141 * We count the total length we send out so we can return it from __vprintf,
142 * since that's what most printf-like functions want to return.
143 */
144static
145void
146__pf_print(PF *pf, const char *txt, size_t len)
147{
148 pf->sendfunc(pf->clientdata, txt, len);
149 pf->charcount += len;
150}

152/*
153 * Reset the state for the next %-field.
154 */
155static
156void
157__pf_endfield(PF *pf)
158{
159 pf->in_pct = 0;
160 pf->size = INTSZ;
161 pf->num = 0;
162 pf->sign = 0;
163 pf->spacing = 0;

165 pf->fillchar = ' ';
166 pf->base = 0;
167 pf->baseprefix = 0;
168 pf->altformat = 0;

171/*
172 * Process modifier chars (between the % and the type specifier)
173 * # use "alternate display format"
174 * - left align in field instead of right align
175 * l value is long (ll = long long)
176 * z value is size_t
177 * 0-9 field width
178 * leading 0 pad with zeros instead of spaces
179 */
180static
181void
182__pf_modifier(PF *pf, int ch)
183{
184 switch (ch) {
185 case '#':
186 pf->altformat = 1;
187 break;
188 case '-':
189 pf->rightspc = 1;
190 break;
191 case 'l':
192 if (pf->size==LONGSZ) {
193#ifdef USE_LONGLONG
194 pf->size = LLONGSZ;
195#endif
196 }
197 else {
198 pf->size = LONGSZ;
199 }
200 break;
201 case 'z':
202 pf->size = SIZETSZ;
203 break;
204 case '0':
205 if (pf->spacing>0) {
206 /*
207 * Already seen some digits; this is part of the
208 * field size.
209 */
210 pf->spacing = pf->spacing*10;
211 }
212 else {
213 /*
214 * Leading zero; set the padding character to 0.
215 */
216 pf->fillchar = '0';
217 }
218 break;
219 default:
220 /*
221 * Invalid characters should be filtered out by a
222 * higher-level function, so if this assert goes off
223 * it's our fault.
224 */
225 assert(ch>'0' && ch<='9');
226
227 /*
228 * Got a digit; accumulate the field size.
229 */
230 pf->spacing = pf->spacing*10 + (ch-'0');
231 break;
232 }
233}

235/*
236 * Retrieve a numeric argument from the argument list and store it
237 * in pf->num, according to the size recorded in pf->size and using
238 * the numeric type specified by ch.
239 */
240static
241void
242__pf_getnum(PF *pf, int ch)
243{
244 if (ch=='p') {
245 /*
246 * Pointer.
247 *
248 * uintptr_t is a C99 standard type that's an unsigned
249 * integer the same size as a pointer.
250 */
251 pf->num = (uintptr_t) va_arg(pf->ap, void *);
252 }
253 else if (ch=='d') {
254 /* signed integer */
255 INTTYPE signednum=0;
256 switch (pf->size) {
257 case INTSZ:
258 /* %d */
259 signednum = va_arg(pf->ap, int);
260 break;
261 case LONGSZ:
262 /* %ld */
263 signednum = va_arg(pf->ap, long);
264 break;
265#ifdef USE_LONGLONG
266 case LLONGSZ:
267 /* %lld */
268 signednum = va_arg(pf->ap, long long);
269 break;
270#endif
271 case SIZETSZ:
272 /* %zd */
273 signednum = va_arg(pf->ap, ssize_t);
274 break;
275 }
276
277 /*
278 * Check for negative numbers.
279 */
280 if (signednum < 0) {
281 pf->sign = -1;
282 pf->num = -signednum;
283 }
284 else {
285 pf->num = signednum;
286 }
287 }
288 else {
289 /* unsigned integer */
290 switch (pf->size) {
291 case INTSZ:
292 /* %u (or %o, %x) */
293 pf->num = va_arg(pf->ap, unsigned int);
294 break;
295 case LONGSZ:
296 /* %lu (or %lo, %lx) */
297 pf->num = va_arg(pf->ap, unsigned long);
298 break;
299#ifdef USE_LONGLONG
300 case LLONGSZ:
301 /* %llu, %llo, %llx */
302 pf->num = va_arg(pf->ap, unsigned long long);
303 break;
304#endif
305 case SIZETSZ:
306 /* %zu, %zo, %zx */
307 pf->num = va_arg(pf->ap, size_t);
308 break;
309 }
310 }
311}

313/*
314 * Set the printing base based on the numeric type specified in ch.
315 * %o octal
316 * %d,%u decimal
317 * %x hex
318 * %p pointer (print as hex)
319 *
320 * If the "alternate format" was requested, or always for pointers,
321 * note to print the C prefix for the type.
322 */
323static
324void
325__pf_setbase(PF *pf, int ch)
326{
327 switch (ch) {
328 case 'd':
329 case 'u':
330 pf->base = 10;
331 break;
332 case 'x':
333 case 'p':
334 pf->base = 16;
335 break;
336 case 'o':
337 pf->base = 8;
338 break;
339 }
340 if (pf->altformat || ch=='p') {
341 pf->baseprefix = 1;
342 }
343}

359/*
360 * General printing function. Prints the string "stuff".
361 * The two prefixes (in practice one is a type prefix, such as "0x",
362 * and the other is the sign) get printed *after* space padding but
363 * *before* zero padding, if padding is on the left.
364 */
365static
366void
367__pf_printstuff(PF *pf,
368 const char *prefix, const char *prefix2,
369 const char *stuff)
370{
371 /* Total length to print. */
372 int len = strlen(prefix)+strlen(prefix2)+strlen(stuff);
373
374 /* Get field width and compute amount of padding in "spc". */
375 int spc = pf->spacing;
376 if (spc > len) {
377 spc -= len;
378 }
379 else {
380 spc = 0;
381 }
382
383 /* If padding on left and the fill char is not 0, pad first. */
384 if (spc > 0 && pf->rightspc==0 && pf->fillchar!='0') {
385 __pf_fill(pf, spc);
386 }
387
388 /* Print the prefixes. */
389 __pf_print(pf, prefix, strlen(prefix));
390 __pf_print(pf, prefix2, strlen(prefix2));
391
392 /* If padding on left and the fill char *is* 0, pad here. */
393 if (spc > 0 && pf->rightspc==0 && pf->fillchar=='0') {
394 __pf_fill(pf, spc);
395 }
396
397 /* Print the actual string. */
398 __pf_print(pf, stuff, strlen(stuff));
399
400 /* If padding on the right, pad afterwards. */
401 if (spc > 0 && pf->rightspc!=0) {
402 __pf_fill(pf, spc);
403 }
404}

406/*
407 * Function to convert a number to ascii and then print it.
408 *
409 * Works from right to left in a buffer of NUMBER_BUF_SIZE bytes.
410 * NUMBER_BUF_SIZE is set so that the longest number string we can
411 * generate (a long long printed in octal) will fit. See above.
412 */
413static
414void
415__pf_printnum(PF *pf)
416{
417 /* Digits to print with. */
418 const char *const digits = "0123456789abcdef";
419
420 char buf[NUMBER_BUF_SIZE]; /* Accumulation buffer for string. */
421 char *x; /* Current pointer into buf. */
422 unsigned INTTYPE xnum; /* Current value to print. */
423 const char *bprefix; /* Base prefix (0, 0x, or nothing) */
424 const char *sprefix; /* Sign prefix (- or nothing) */
425
426 /* Start in the last slot of the buffer. */
427 x = buf+sizeof(buf)-1;
428
429 /* Insert null terminator. */
430 *x-- = 0;
431
432 /* Initialize value. */
433 xnum = pf->num;
434
435 /*
436 * Convert a single digit.
437 * Do this loop at least once - that way 0 prints as 0 and not "".
438 */
439 do {
440 /*
441 * Get the digit character for the least significant
442 * part of xnum.
443 */
444 *x = digits[xnum % pf->base];
445
446 /*
447 * Back up the pointer to point to the next space to the left.
448 */
449 x--;
450
451 /*
452 * Drop the value of the digit we just printed from xnum.
453 */
454 xnum = xnum / pf->base;
455
456 /*
457 * If xnum hits 0 there's no more number left.
458 */
459 } while (xnum > 0);
460
461 /*
462 * x points to the *next* slot in the buffer to use.
463 * However, we're done printing the number. So it's pointing
464 * one slot *before* the start of the actual number text.
465 * So advance it by one so it actually points at the number.
466 */
467 x++;
468
469 /*
470 * If a base prefix was requested, select it.
471 */
472 if (pf->baseprefix && pf->base==16) {
473 bprefix = "0x";
474 }
475 else if (pf->baseprefix && pf->base==8) {
476 bprefix = "0";
477 }
478 else {
479 bprefix = "";
480 }
481
482 /*
483 * Choose the sign prefix.
484 */
485 sprefix = pf->sign ? "-" : "";
486
487 /*
488 * Now actually print the string we just generated.
489 */
490 __pf_printstuff(pf, sprefix, bprefix, x);
491}

493/*
494 * Process a single character out of the format string.
495 */
496static
497void
498__pf_send(PF *pf, int ch)
499{
500 /* Cannot get NULs here. */
501 assert(ch!=0);
502
503 if (pf->in_pct==0 && ch!='%') {
504 /*
505 * Not currently in a format, and not a %. Just send
506 * the character on through.
507 */
508 char c = ch;
509 __pf_print(pf, &c, 1);
510 }
511 else if (pf->in_pct==0) {
512 /*
513 * Not in a format, but got a %. Start a format.
514 */
515 pf->in_pct = 1;
516 }
517 else if (strchr("#-lz0123456789", ch)) {
518 /*
519 * These are the modifier characters we recognize.
520 * (These are the characters between the % and the type.)
521 */
522 __pf_modifier(pf, ch);
523 }
524 else if (strchr("doupx", ch)) {
525 /*
526 * Integer types.
527 * Fetch the number, set the base, print it, then
528 * reset for the next format.
529 */
530 __pf_getnum(pf, ch);
531 __pf_setbase(pf, ch);
532 __pf_printnum(pf);
533 __pf_endfield(pf);
534 }
535 else if (ch=='s') {
536 /*
537 * Print a string.
538 */
539 const char *str = va_arg(pf->ap, const char *);
540 if (str==NULL) {
541 str = "(null)";
542 }
543 __pf_printstuff(pf, "", "", str);
544 __pf_endfield(pf);
545 }
546 else {
547 /*
548 * %%, %c, or illegal character.
549 * Illegal characters are printed like %%.
550 * for example, %5k prints " k".
551 */
552 char x[2];
553 if (ch=='c') {
554 x[0] = va_arg(pf->ap, int);
555 }
556 else {
557 x[0] = ch;
558 }
559 x[1] = 0;
560 __pf_printstuff(pf, "", "", x);
561 __pf_endfield(pf);
562 }
563}

565/*
566 * Do a whole printf session.
567 * Create and initialize a printf state object,
568 * then send it each character from the format string.
569 */
570int
571__vprintf(void (*func)(void *clientdata, const char *str, size_t len),
572 void *clientdata, const char *format, va_list ap)
573{
574 PF pf;
575 int i;
576
577 pf.sendfunc = func;
578 pf.clientdata = clientdata;
579#ifdef va_copy
580 va_copy(pf.ap, ap);
581#else
582 pf.ap = ap;
583#endif
584 pf.charcount = 0;
585 __pf_endfield(&pf);
586
587 for (i=0; format[i]; i++) {
588 __pf_send(&pf, format[i]);
589 }
590
591 return pf.charcount;
592}
