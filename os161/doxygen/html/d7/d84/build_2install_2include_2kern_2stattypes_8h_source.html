30#ifndef _KERN_STATTYPES_H_
31#define _KERN_STATTYPES_H_
32
33/*
34 * Further supporting material for stat(), fstat(), and lstat().
35 *
36 * File types for st_mode. (The permissions are the low 12 bits.)
37 *
38 * These are also used, shifted right by those 12 bits, in struct
39 * dirent in libc, which is why they get their own file.
40 *
41 * Non-underscore versions of the names can be gotten from <stat.h>
42 * (kernel) or <sys/stat.h> (userland).
43 */
44
45#define _S_IFMT 070000 /* mask for type of file */
46#define _S_IFREG 010000 /* ordinary regular file */
47#define _S_IFDIR 020000 /* directory */
48#define _S_IFLNK 030000 /* symbolic link */
49#define _S_IFIFO 040000 /* pipe or named pipe */
50#define _S_IFSOCK 050000 /* socket */
51#define _S_IFCHR 060000 /* character device */
52#define _S_IFBLK 070000 /* block device */
53
54
55#endif /* _KERN_STATTYPES_H_ */
