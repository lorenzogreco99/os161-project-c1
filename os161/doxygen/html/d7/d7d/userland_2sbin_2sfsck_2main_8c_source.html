45static int badness=0;
46
47/*
48 * Update the badness state. (codes are in main.h)
49 *
50 * The badness state only gets worse, and is ultimately the process
51 * exit code.
52 */
53void
54setbadness(int code)
55{
56 if (badness < code) {
57 badness = code;
58 }
59}

61/*
62 * Main.
63 */
64int
65main(int argc, char **argv)
66{
67#ifdef HOST
68 hostcompat_init(argc, argv);
69#endif
70
71 /* FUTURE: add -n option */
72 if (argc!=2) {
73 errx(EXIT_USAGE, "Usage: sfsck device/diskfile");
74 }
75
76 opendisk(argv[1]);
77
78 sfs_setup();
79 sb_load();
80 sb_check();
81 freemap_setup();
82
83 printf("Phase 1 -- check blocks and sizes\n");
84 pass1();
85 freemap_check();
86
87 printf("Phase 2 -- check directory tree\n");
88 inode_sorttable();
89 pass2();
90
91 printf("Phase 3 -- check reference counts\n");
92 inode_adjust_filelinks();
93
94 closedisk();
95
96 warnx("%lu blocks used (of %lu); %lu directories; %lu files",
97 freemap_blocksused(), (unsigned long)sb_totalblocks(),
98 pass1_founddirs(), pass1_foundfiles());
99
100 switch (badness) {
101 case EXIT_USAGE:
102 case EXIT_FATAL:
103 default:
104 /* not supposed to happen here */
105 assert(0);
106 break;
107 case EXIT_UNRECOV:
108 warnx("WARNING - unrecoverable errors. Maybe try again?");
109 break;
110 case EXIT_RECOV:
111 warnx("Caution - filesystem modified. Run again for luck.");
112 break;
113 case EXIT_CLEAN:
114 break;
115 }
116
117 return badness;
118}
