49/*
50 * Basic spinlock.
51 *
52 * Note that spinlocks are held by CPUs, not by threads.
53 *
54 * This structure is made public so spinlocks do not have to be
55 * malloc'd; however, code that uses spinlocks should not look inside
56 * the structure directly but always use the spinlock API functions.
57 */
58struct spinlock {
59 volatile spinlock_data_t splk_lock; /* Memory word where we spin. */
60 struct cpu *splk_holder; /* CPU holding this lock. */
61 HANGMAN_LOCKABLE(splk_hangman); /* Deadlock detector hook. */
62};

74/*
75 * Spinlock functions.
76 *
77 * init Initialize the contents of a spinlock.
78 * cleanup Opposite of init. Lock must be unlocked.
79 *
80 * acquire Get the lock, spinning as necessary. Also disables interrupts.
81 * release Release the lock. May re-enable interrupts.
82 *
83 * do_i_hold Check if the current CPU holds the lock.
84 */
85

90void spinlock_release(struct spinlock *lk);
