177static
178void
179common_badbuf(struct buftest *info, void *buf, const char *bufdesc)
180{
181 int rv;
182
183
184 report_begin("%s with %s buffer", info->name, bufdesc);
185 info->setup();
186 rv = info->op(buf);
187 report_check(rv, errno, EFAULT);
188 info->cleanup();
189}

191static
192void
193any_badbuf(struct buftest *info)
194{
195 common_badbuf(info, NULL, "NULL");
196 common_badbuf(info, INVAL_PTR, "invalid");
197 common_badbuf(info, KERN_PTR, "kernel-space");
198}
