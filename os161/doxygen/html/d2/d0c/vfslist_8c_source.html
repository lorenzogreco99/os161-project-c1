47/*
48 * Structure for a single named device.
49 *
50 * kd_name - Name of device (eg, "lhd0"). Should always be set to
51 * a valid string.
52 *
53 * kd_rawname - Name of raw device (eg, "lhd0raw"). Is non-NULL if and
54 * only if this device can have a filesystem mounted on
55 * it.
56 *
57 * kd_device - Device object this name refers to. May be NULL if kd_fs
58 * is hardwired.
59 *
60 * kd_fs - Filesystem object mounted on, or associated with, this
61 * device. NULL if there is no filesystem.
62 *
63 * A filesystem can be associated with a device without having been
64 * mounted if the device was created that way. In this case,
65 * kd_rawname is NULL (prohibiting mount/unmount), and, as there is
66 * then no way to access kd_device, it will be NULL as well. This is
67 * intended for devices that are inherently filesystems, like emu0.
68 *
69 * Referencing kd_name, or the filesystem volume name, on a device
70 * with a filesystem mounted returns the root of the filesystem.
71 * Referencing kd_name on a mountable device with no filesystem
72 * returns ENXIO. Referencing kd_name on a device that is not
73 * mountable and has no filesystem, or kd_rawname on a mountable
74 * device, returns the device itself.
75 */
76
77struct knowndev {
78 char *kd_name;
79 char *kd_rawname;
80 struct device *kd_device;
81 struct vnode *kd_vnode;
82 struct fs *kd_fs;
83};
84
85/* A placeholder for kd_fs for devices used as swap */
86#define SWAP_FS ((struct fs *)-1)

119/*
120 * Operations on vfs_biglock. We make it recursive to avoid having to
121 * think about where we do and don't already hold it. This is an
122 * undesirable hack that's frequently necessary when a lock covers too
123 * much material. Your solution scheme for FS and VFS locking should
124 * not require recursive locks.
125 */
126void
127vfs_biglock_acquire(void)
128{
129 if (!lock_do_i_hold(vfs_biglock)) {
130 lock_acquire(vfs_biglock);
131 }
132 else if (vfs_biglock_depth == 0) {
133 /*
134 * Supposedly we hold it, but the depth is 0. This may
135 * mean: (1) the count is messed up, or (2)
136 * lock_do_i_hold is lying. Since OS/161 ships out of
137 * the box with unimplemented locks (students
138 * implement them) that always return true, assume
139 * situation (2). In this case acquire the lock
140 * anyway.
141 *
142 * Once you have working locks, this won't be the
143 * case, and if you get here it should be situation
144 * (1), in which case the count is messed up and one
145 * can panic.
146 */
147 lock_acquire(vfs_biglock);
148 }
149 vfs_biglock_depth++;
150}
151
152void
153vfs_biglock_release(void)
154{
155 KASSERT(lock_do_i_hold(vfs_biglock));
156 KASSERT(vfs_biglock_depth > 0);
157 vfs_biglock_depth--;
158 if (vfs_biglock_depth == 0) {
159 lock_release(vfs_biglock);
160 }
161}
162
163bool
164vfs_biglock_do_i_hold(void)
165{
166 return lock_do_i_hold(vfs_biglock);
167}

169/*
170 * Global sync function - call FSOP_SYNC on all devices.
171 */
172int
173vfs_sync(void)
174{
175 struct knowndev *dev;
176 unsigned i, num;
177
178 vfs_biglock_acquire();
179
180 num = knowndevarray_num(knowndevs);
181 for (i=0; i<num; i++) {
182 dev = knowndevarray_get(knowndevs, i);
183 if (dev->kd_fs != NULL && dev->kd_fs != SWAP_FS) {
184 /*result =*/ FSOP_SYNC(dev->kd_fs);
185 }
186 }
187
188 vfs_biglock_release();
189
190 return 0;
191}

193/*
194 * Given a device name (lhd0, emu0, somevolname, null, etc.), hand
195 * back an appropriate vnode.
196 */
197int
198vfs_getroot(const char *devname, struct vnode **ret)
199{
200 struct knowndev *kd;
201 unsigned i, num;
202
203 KASSERT(vfs_biglock_do_i_hold());
204
205 num = knowndevarray_num(knowndevs);
206 for (i=0; i<num; i++) {
207 kd = knowndevarray_get(knowndevs, i);
208
209 /*
210 * If this device has a mounted filesystem, and
211 * DEVNAME names either the filesystem or the device,
212 * return the root of the filesystem.
213 *
214 * If it has no mounted filesystem, it's mountable,
215 * and DEVNAME names the device, return ENXIO.
216 */
217
218 if (kd->kd_fs != NULL && kd->kd_fs != SWAP_FS) {
219 const char *volname;
220 volname = FSOP_GETVOLNAME(kd->kd_fs);
221
222 if (!strcmp(kd->kd_name, devname) ||
223 (volname!=NULL && !strcmp(volname, devname))) {
224 return FSOP_GETROOT(kd->kd_fs, ret);
225 }
226 }
227 else {
228 if (kd->kd_rawname!=NULL &&
229 !strcmp(kd->kd_name, devname)) {
230 return ENXIO;
231 }
232 }
233
234 /*
235 * If DEVNAME names the device, and we get here, it
236 * must have no fs and not be mountable. In this case,
237 * we return the device itself.
238 */
239 if (!strcmp(kd->kd_name, devname)) {
240 KASSERT(kd->kd_fs==NULL);
241 KASSERT(kd->kd_rawname==NULL);
242 KASSERT(kd->kd_device != NULL);
243 VOP_INCREF(kd->kd_vnode);
244 *ret = kd->kd_vnode;
245 return 0;
246 }
247
248 /*
249 * If the device has a rawname and DEVNAME names that,
250 * return the device itself.
251 */
252 if (kd->kd_rawname!=NULL && !strcmp(kd->kd_rawname, devname)) {
253 KASSERT(kd->kd_device != NULL);
254 VOP_INCREF(kd->kd_vnode);
255 *ret = kd->kd_vnode;
256 return 0;
257 }
258
259 /*
260 * If none of the above tests matched, we didn't name
261 * any of the names of this device, so go on to the
262 * next one.
263 */
264 }
265
266 /*
267 * If we got here, the device specified by devname doesn't exist.
268 */
269
270 return ENODEV;
271}

273/*
274 * Given a filesystem, hand back the name of the device it's mounted on.
275 */
276const char *
277vfs_getdevname(struct fs *fs)
278{
279 struct knowndev *kd;
280 unsigned i, num;
281
282 KASSERT(fs != NULL);
283
284 KASSERT(vfs_biglock_do_i_hold());
285
286 num = knowndevarray_num(knowndevs);
287 for (i=0; i<num; i++) {
288 kd = knowndevarray_get(knowndevs, i);
289
290 if (kd->kd_fs == fs) {
291 /*
292 * This is not a race condition: as long as the
293 * guy calling us holds a reference to the fs,
294 * the fs cannot go away, and the device can't
295 * go away until the fs goes away.
296 */
297 return kd->kd_name;
298 }
299 }
300
301 return NULL;
302}

321/*
322 * Check if the two strings passed in are the same, if they're both
323 * not NULL (the latter part being significant).
324 */
325static
326inline
327int
328samestring(const char *a, const char *b)
329{
330 if (a==NULL || b==NULL) {
331 return 0;
332 }
333 return !strcmp(a, b);
334}

336/*
337 * Check if the first string passed is the same as any of the three others,
338 * if they're not NULL.
339 */
340static
341inline
342int
343samestring3(const char *a, const char *b, const char *c, const char *d)
344{
345 return samestring(a,b) || samestring(a,c) || samestring(a,d);
346}

348/*
349 * Check if any of the three names passed in already exists as a device
350 * name.
351 */
352
353static
354int
355badnames(const char *n1, const char *n2, const char *n3)
356{
357 const char *volname;
358 unsigned i, num;
359 struct knowndev *kd;
360
361 KASSERT(vfs_biglock_do_i_hold());
362
363 num = knowndevarray_num(knowndevs);
364 for (i=0; i<num; i++) {
365 kd = knowndevarray_get(knowndevs, i);
366
367 if (kd->kd_fs != NULL && kd->kd_fs != SWAP_FS) {
368 volname = FSOP_GETVOLNAME(kd->kd_fs);
369 if (samestring3(volname, n1, n2, n3)) {
370 return 1;
371 }
372 }
373
374 if (samestring3(kd->kd_rawname, n1, n2, n3) ||
375 samestring3(kd->kd_name, n1, n2, n3)) {
376 return 1;
377 }
378 }
379
380 return 0;
381}

383/*
384 * Add a new device to the VFS layer's device table.
385 *
386 * If "mountable" is set, the device will be treated as one that expects
387 * to have a filesystem mounted on it, and a raw device will be created
388 * for direct access.
389 */
390static
391int
392vfs_doadd(const char *dname, int mountable, struct device *dev, struct fs *fs)
393{
394 char *name=NULL, *rawname=NULL;
395 struct knowndev *kd=NULL;
396 struct vnode *vnode=NULL;
397 const char *volname=NULL;
398 unsigned index;
399 int result;
400
401 /* Silence warning with gcc 4.8 -Og (but not -O2) */
402 index = 0;
403
404 vfs_biglock_acquire();
405
406 name = kstrdup(dname);
407 if (name==NULL) {
408 result = ENOMEM;
409 goto fail;
410 }
411 if (mountable) {
412 rawname = mkrawname(name);
413 if (rawname==NULL) {
414 result = ENOMEM;
415 goto fail;
416 }
417 }
418
419 vnode = dev_create_vnode(dev);
420 if (vnode==NULL) {
421 result = ENOMEM;
422 goto fail;
423 }
424
425 kd = kmalloc(sizeof(struct knowndev));
426 if (kd==NULL) {
427 result = ENOMEM;
428 goto fail;
429 }
430
431 kd->kd_name = name;
432 kd->kd_rawname = rawname;
433 kd->kd_device = dev;
434 kd->kd_vnode = vnode;
435 kd->kd_fs = fs;
436
437 if (fs!=NULL) {
438 volname = FSOP_GETVOLNAME(fs);
439 }
440
441 if (badnames(name, rawname, volname)) {
442 result = EEXIST;
443 goto fail;
444 }
445
446 result = knowndevarray_add(knowndevs, kd, &index);
447 if (result) {
448 goto fail;
449 }
450
451 if (dev != NULL) {
452 /* use index+1 as the device number, so 0 is reserved */
453 dev->d_devnumber = index+1;
454 }
455
456 vfs_biglock_release();
457 return 0;
458
459 fail:
460 if (name) {
461 kfree(name);
462 }
463 if (rawname) {
464 kfree(rawname);
465 }
466 if (vnode) {
467 dev_uncreate_vnode(vnode);
468 }
469 if (kd) {
470 kfree(kd);
471 }
472
473 vfs_biglock_release();
474 return result;
475}

500/*
501 * Look for a mountable device named DEVNAME.
502 * Should already hold knowndevs_lock.
503 */
504static
505int
506findmount(const char *devname, struct knowndev **result)
507{
508 struct knowndev *dev;
509 unsigned i, num;
510 bool found = false;
511
512 KASSERT(vfs_biglock_do_i_hold());
513
514 num = knowndevarray_num(knowndevs);
515 for (i=0; !found && i<num; i++) {
516 dev = knowndevarray_get(knowndevs, i);
517 if (dev->kd_rawname==NULL) {
518 /* not mountable/unmountable */
519 continue;
520 }
521
522 if (!strcmp(devname, dev->kd_name)) {
523 *result = dev;
524 found = true;
525 }
526 }
527
528 return found ? 0 : ENODEV;
529}

531/*
532 * Mount a filesystem. Once we've found the device, call MOUNTFUNC to
533 * set up the filesystem and hand back a struct fs.
534 *
535 * The DATA argument is passed through unchanged to MOUNTFUNC.
536 */
537int
538vfs_mount(const char *devname, void *data,
539 int (*mountfunc)(void *data, struct device *, struct fs **ret))
540{
541 const char *volname;
542 struct knowndev *kd;
543 struct fs *fs;
544 int result;
545
546 vfs_biglock_acquire();
547
548 result = findmount(devname, &kd);
549 if (result) {
550 vfs_biglock_release();
551 return result;
552 }
553
554 if (kd->kd_fs != NULL) {
555 vfs_biglock_release();
556 return EBUSY;
557 }
558 KASSERT(kd->kd_rawname != NULL);
559 KASSERT(kd->kd_device != NULL);
560
561 result = mountfunc(data, kd->kd_device, &fs);
562 if (result) {
563 vfs_biglock_release();
564 return result;
565 }
566
567 KASSERT(fs != NULL);
568 KASSERT(fs != SWAP_FS);
569
570 kd->kd_fs = fs;
571
572 volname = FSOP_GETVOLNAME(fs);
573 kprintf("vfs: Mounted %s: on %s\n",
574 volname ? volname : kd->kd_name, kd->kd_name);
575
576 vfs_biglock_release();
577 return 0;
578}

580/*
581 * Like mount, but for attaching swap. Hands back the raw device
582 * vnode. Unlike mount tolerates a trailing colon on the device name,
583 * to avoid student-facing confusion.
584 */
585int
586vfs_swapon(const char *devname, struct vnode **ret)
587{
588 char *myname = NULL;
589 size_t len;
590 struct knowndev *kd;
591 int result;
592
593 len = strlen(devname);
594 if (len > 0 && devname[len - 1] == ':') {
595 /* tolerate trailing :, e.g. lhd0: rather than lhd0 */
596 myname = kstrdup(devname);
597 if (myname == NULL) {
598 return ENOMEM;
599 }
600 myname[len - 1] = 0;
601 devname = myname;
602 }
603
604 vfs_biglock_acquire();
605
606 result = findmount(devname, &kd);
607 if (result) {
608 goto out;
609 }
610
611 if (kd->kd_fs != NULL) {
612 result = EBUSY;
613 goto out;
614 }
615 KASSERT(kd->kd_rawname != NULL);
616 KASSERT(kd->kd_device != NULL);
617
618 kprintf("vfs: Swap attached to %s\n", kd->kd_name);
619
620 kd->kd_fs = SWAP_FS;
621 VOP_INCREF(kd->kd_vnode);
622 *ret = kd->kd_vnode;
623
624 out:
625 vfs_biglock_release();
626 if (myname != NULL) {
627 kfree(myname);
628 }
629
630 return result;
631}

633/*
634 * Unmount a filesystem/device by name.
635 * First calls FSOP_SYNC on the filesystem; then calls FSOP_UNMOUNT.
636 */
637int
638vfs_unmount(const char *devname)
639{
640 struct knowndev *kd;
641 int result;
642
643 vfs_biglock_acquire();
644
645 result = findmount(devname, &kd);
646 if (result) {
647 goto fail;
648 }
649
650 if (kd->kd_fs == NULL || kd->kd_fs == SWAP_FS) {
651 result = EINVAL;
652 goto fail;
653 }
654 KASSERT(kd->kd_rawname != NULL);
655 KASSERT(kd->kd_device != NULL);
656
657 /* sync the fs */
658 result = FSOP_SYNC(kd->kd_fs);
659 if (result) {
660 goto fail;
661 }
662
663 result = FSOP_UNMOUNT(kd->kd_fs);
664 if (result) {
665 goto fail;
666 }
667
668 kprintf("vfs: Unmounted %s:\n", kd->kd_name);
669
670 /* now drop the filesystem */
671 kd->kd_fs = NULL;
672
673 KASSERT(result==0);
674
675 fail:
676 vfs_biglock_release();
677 return result;
678}

717/*
718 * Global unmount function.
719 */
720int
721vfs_unmountall(void)
722{
723 struct knowndev *dev;
724 unsigned i, num;
725 int result;
726
727 vfs_biglock_acquire();
728
729 num = knowndevarray_num(knowndevs);
730 for (i=0; i<num; i++) {
731 dev = knowndevarray_get(knowndevs, i);
732 if (dev->kd_rawname == NULL) {
733 /* not mountable/unmountable */
734 continue;
735 }
736 if (dev->kd_fs == NULL) {
737 /* not mounted */
738 continue;
739 }
740 if (dev->kd_fs == SWAP_FS) {
741 /* just drop it */
742 dev->kd_fs = NULL;
743 continue;
744 }
745
746 kprintf("vfs: Unmounting %s:\n", dev->kd_name);
747
748 result = FSOP_SYNC(dev->kd_fs);
749 if (result) {
750 kprintf("vfs: Warning: sync failed for %s: %s, trying "
751 "again\n", dev->kd_name, strerror(result));
752
753 result = FSOP_SYNC(dev->kd_fs);
754 if (result) {
755 kprintf("vfs: Warning: sync failed second time"
756 " for %s: %s, giving up...\n",
757 dev->kd_name, strerror(result));
758 /*
759 * Do not attempt to complete the
760 * unmount as it will likely explode.
761 */
762 continue;
763 }
764 }
765
766 result = FSOP_UNMOUNT(dev->kd_fs);
767 if (result == EBUSY) {
768 kprintf("vfs: Cannot unmount %s: (busy)\n",
769 dev->kd_name);
770 continue;
771 }
772 if (result) {
773 kprintf("vfs: Warning: unmount failed for %s:"
774 " %s, already synced, dropping...\n",
775 dev->kd_name, strerror(result));
776 continue;
777 }
778
779 /* now drop the filesystem */
780 dev->kd_fs = NULL;
781 }
782
783 vfs_biglock_release();
784
785 return 0;
786}
