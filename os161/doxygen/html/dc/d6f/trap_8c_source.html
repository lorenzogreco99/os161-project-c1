69/*
70 * Function called when user-level code hits a fatal fault.
71 */
72static
73void
74kill_curthread(vaddr_t epc, unsigned code, vaddr_t vaddr)
75{
76 int sig = 0;
77
78 KASSERT(code < NTRAPCODES);
79 switch (code) {
80 case EX_IRQ:
81 case EX_IBE:
82 case EX_DBE:
83 case EX_SYS:
84 /* should not be seen */
85 KASSERT(0);
86 sig = SIGABRT;
87 break;
88 case EX_MOD:
89 case EX_TLBL:
90 case EX_TLBS:
91 sig = SIGSEGV;
92 break;
93 case EX_ADEL:
94 case EX_ADES:
95 sig = SIGBUS;
96 break;
97 case EX_BP:
98 sig = SIGTRAP;
99 break;
100 case EX_RI:
101 sig = SIGILL;
102 break;
103 case EX_CPU:
104 sig = SIGSEGV;
105 break;
106 case EX_OVF:
107 sig = SIGFPE;
108 break;
109 }
110
111 /*
112 * You will probably want to change this.
113 */
114
115 kprintf("Fatal user mode trap %u sig %d (%s, epc 0x%x, vaddr 0x%x)\n",
116 code, sig, trapcodenames[code], epc, vaddr);
117 panic("I don't know how to handle this\n");
118}

120/*
121 * General trap (exception) handling function for mips.
122 * This is called by the assembly-language exception handler once
123 * the trapframe has been set up.
124 */
125void
126mips_trap(struct trapframe *tf)
127{
128 uint32_t code;
129 /*bool isutlb; -- not used */
130 bool iskern;
131 int spl;
132
133 /* The trap frame is supposed to be 35 registers long. */
134 KASSERT(sizeof(struct trapframe)==(35*4));
135
136 /*
137 * Extract the exception code info from the register fields.
138 */
139 code = (tf->tf_cause & CCA_CODE) >> CCA_CODESHIFT;
140 /*isutlb = (tf->tf_cause & CCA_UTLB) != 0;*/
141 iskern = (tf->tf_status & CST_KUp) == 0;
142
143 KASSERT(code < NTRAPCODES);
144
145 /* Make sure we haven't run off our stack */
146 if (curthread != NULL && curthread->t_stack != NULL) {
147 KASSERT((vaddr_t)tf > (vaddr_t)curthread->t_stack);
148 KASSERT((vaddr_t)tf < (vaddr_t)(curthread->t_stack
149 + STACK_SIZE));
150 }
151
152 /* Interrupt? Call the interrupt handler and return. */
153 if (code == EX_IRQ) {
154 int old_in;
155 bool doadjust;
156
157 old_in = curthread->t_in_interrupt;
158 curthread->t_in_interrupt = 1;
159
160 /*
161 * The processor has turned interrupts off; if the
162 * currently recorded interrupt state is interrupts on
163 * (spl of 0), adjust the recorded state to match, and
164 * restore after processing the interrupt.
165 *
166 * How can we get an interrupt if the recorded state
167 * is interrupts off? Well, as things currently stand
168 * when the CPU finishes idling it flips interrupts on
169 * and off to allow things to happen, but leaves
170 * curspl high while doing so.
171 *
172 * While we're here, assert that the interrupt
173 * handling code hasn't leaked a spinlock or an
174 * splhigh().
175 */
176
177 if (curthread->t_curspl == 0) {
178 KASSERT(curthread->t_curspl == 0);
179 KASSERT(curthread->t_iplhigh_count == 0);
180 curthread->t_curspl = IPL_HIGH;
181 curthread->t_iplhigh_count++;
182 doadjust = true;
183 }
184 else {
185 doadjust = false;
186 }
187
188 mainbus_interrupt(tf);
189
190 if (doadjust) {
191 KASSERT(curthread->t_curspl == IPL_HIGH);
192 KASSERT(curthread->t_iplhigh_count == 1);
193 curthread->t_iplhigh_count--;
194 curthread->t_curspl = 0;
195 }
196
197 curthread->t_in_interrupt = old_in;
198 goto done2;
199 }
200
201 /*
202 * The processor turned interrupts off when it took the trap.
203 *
204 * While we're in the kernel, and not actually handling an
205 * interrupt, restore the interrupt state to where it was in
206 * the previous context, which may be low (interrupts on).
207 *
208 * Do this by forcing splhigh(), which may do a redundant
209 * cpu_irqoff() but forces the stored MI interrupt state into
210 * sync, then restoring the previous state.
211 */
212 spl = splhigh();
213 splx(spl);
214
215 /* Syscall? Call the syscall handler and return. */
216 if (code == EX_SYS) {
217 /* Interrupts should have been on while in user mode. */
218 KASSERT(curthread->t_curspl == 0);
219 KASSERT(curthread->t_iplhigh_count == 0);
220
221 DEBUG(DB_SYSCALL, "syscall: #%d, args %x %x %x %x\n",
222 tf->tf_v0, tf->tf_a0, tf->tf_a1, tf->tf_a2, tf->tf_a3);
223
224 syscall(tf);
225 goto done;
226 }
227
228 /*
229 * Ok, it wasn't any of the really easy cases.
230 * Call vm_fault on the TLB exceptions.
231 * Panic on the bus error exceptions.
232 */
233 switch (code) {
234 case EX_MOD:
235 if (vm_fault(VM_FAULT_READONLY, tf->tf_vaddr)==0) {
236 goto done;
237 }
238 break;
239 case EX_TLBL:
240 if (vm_fault(VM_FAULT_READ, tf->tf_vaddr)==0) {
241 goto done;
242 }
243 break;
244 case EX_TLBS:
245 if (vm_fault(VM_FAULT_WRITE, tf->tf_vaddr)==0) {
246 goto done;
247 }
248 break;
249 case EX_IBE:
250 case EX_DBE:
251 /*
252 * This means you loaded invalid TLB entries, or
253 * touched invalid parts of the direct-mapped
254 * segments. These are serious kernel errors, so
255 * panic.
256 *
257 * The MIPS won't even tell you what invalid address
258 * caused the bus error.
259 */
260 panic("Bus error exception, PC=0x%x\n", tf->tf_epc);
261 break;
262 }
263
264 /*
265 * If we get to this point, it's a fatal fault - either it's
266 * one of the other exceptions, like illegal instruction, or
267 * it was a page fault we couldn't handle.
268 */
269
270 if (!iskern) {
271 /*
272 * Fatal fault in user mode.
273 * Kill the current user process.
274 */
275 kill_curthread(tf->tf_epc, code, tf->tf_vaddr);
276 goto done;
277 }
278
279 /*
280 * Fatal fault in kernel mode.
281 *
282 * If pcb_badfaultfunc is set, we do not panic; badfaultfunc is
283 * set by copyin/copyout and related functions to signify that
284 * the addresses they're accessing are userlevel-supplied and
285 * not trustable. What we actually want to do is resume
286 * execution at the function pointed to by badfaultfunc. That's
287 * going to be "copyfail" (see copyinout.c), which longjmps
288 * back to copyin/copyout or wherever and returns EFAULT.
289 *
290 * Note that we do not just *call* this function, because that
291 * won't necessarily do anything. We want the control flow
292 * that is currently executing in copyin (or whichever), and
293 * is stopped while we process the exception, to *teleport* to
294 * copyfail.
295 *
296 * This is accomplished by changing tf->tf_epc and returning
297 * from the exception handler.
298 */
299
300 if (curthread != NULL &&
301 curthread->t_machdep.tm_badfaultfunc != NULL) {
302 tf->tf_epc = (vaddr_t) curthread->t_machdep.tm_badfaultfunc;
303 goto done;
304 }
305
306 /*
307 * Really fatal kernel-mode fault.
308 */
309
310 kprintf("panic: Fatal exception %u (%s) in kernel mode\n", code,
311 trapcodenames[code]);
312 kprintf("panic: EPC 0x%x, exception vaddr 0x%x\n",
313 tf->tf_epc, tf->tf_vaddr);
314
315 panic("I can't handle this... I think I'll just die now...\n");
316
317 done:
318 /*
319 * Turn interrupts off on the processor, without affecting the
320 * stored interrupt state.
321 */
322 cpu_irqoff();
323 done2:
324
325 /*
326 * The boot thread can get here (e.g. on interrupt return) but
327 * since it doesn't go to userlevel, it can't be returning to
328 * userlevel, so there's no need to set cputhreads[] and
329 * cpustacks[]. Just return.
330 */
331 if (curthread->t_stack == NULL) {
332 return;
333 }
334
335 cputhreads[curcpu->c_number] = (vaddr_t)curthread;
336 cpustacks[curcpu->c_number] = (vaddr_t)curthread->t_stack + STACK_SIZE;
337
338 /*
339 * This assertion will fail if either
340 * (1) curthread->t_stack is corrupted, or
341 * (2) the trap frame is somehow on the wrong kernel stack.
342 *
343 * If cpustacks[] is corrupted, the next trap back to the
344 * kernel will (most likely) hang the system, so it's better
345 * to find out now.
346 */
347 KASSERT(SAME_STACK(cpustacks[curcpu->c_number]-1, (vaddr_t)tf));
348}

350/*
351 * Function for entering user mode.
352 *
353 * This should not be used by threads returning from traps - they
354 * should just return from mips_trap(). It should be used by threads
355 * entering user mode for the first time - whether the child thread in
356 * a fork(), or into a brand-new address space after exec(), or when
357 * starting the first userlevel program.
358 *
359 * It works by jumping into the exception return code.
360 *
361 * mips_usermode is common code for this. It cannot usefully be called
362 * outside the mips port, but should be called from one of the
363 * following places:
364 * - enter_new_process, for use by exec and equivalent.
365 * - enter_forked_process, in syscall.c, for use by fork.
366 */
367void
368mips_usermode(struct trapframe *tf)
369{
370
371 /*
372 * Interrupts should be off within the kernel while entering
373 * user mode. However, while in user mode, interrupts should
374 * be on. To interact properly with the spl-handling logic
375 * above, we explicitly call spl0() and then call cpu_irqoff().
376 */
377 spl0();
378 cpu_irqoff();
379
380 cputhreads[curcpu->c_number] = (vaddr_t)curthread;
381 cpustacks[curcpu->c_number] = (vaddr_t)curthread->t_stack + STACK_SIZE;
382
383 /*
384 * This assertion will fail if either
385 * (1) cpustacks[] is corrupted, or
386 * (2) the trap frame is not on our own kernel stack, or
387 * (3) the boot thread tries to enter user mode.
388 *
389 * If cpustacks[] is corrupted, the next trap back to the
390 * kernel will (most likely) hang the system, so it's better
391 * to find out now.
392 *
393 * It's necessary for the trap frame used here to be on the
394 * current thread's own stack. It cannot correctly be on
395 * either another thread's stack or in the kernel heap.
396 * (Exercise: why?)
397 */
398 KASSERT(SAME_STACK(cpustacks[curcpu->c_number]-1, (vaddr_t)tf));
399
400 /*
401 * This actually does it. See exception-*.S.
402 */
403 asm_usermode(tf);
404}

406/*
407 * enter_new_process: go to user mode after loading an executable.
408 *
409 * Performs the necessary initialization so that the user program will
410 * get the arguments supplied in argc/argv (note that argv must be a
411 * user-level address) and the environment pointer env (ditto), and
412 * begin executing at the specified entry point. The stack pointer is
413 * initialized from the stackptr argument. Note that passing argc/argv
414 * may use additional stack space on some other platforms (but not on
415 * mips).
416 *
417 * Unless you implement execve() that passes environments around, just
418 * pass NULL for the environment.
419 *
420 * Works by creating an ersatz trapframe.
421 */
422void
423enter_new_process(int argc, userptr_t argv, userptr_t env,
424 vaddr_t stack, vaddr_t entry)
425{
426 struct trapframe tf;
427
428 bzero(&tf, sizeof(tf));
429
430 tf.tf_status = CST_IRQMASK | CST_IEp | CST_KUp;
431 tf.tf_epc = entry;
432 tf.tf_a0 = argc;
433 tf.tf_a1 = (vaddr_t)argv;
434 tf.tf_a2 = (vaddr_t)env;
435 tf.tf_sp = stack;
436
437 mips_usermode(&tf);
438}
