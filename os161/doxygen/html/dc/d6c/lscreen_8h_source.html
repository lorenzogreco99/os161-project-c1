33/*
34 * Hardware device data for memory-mapped fullscreen text console.
35 */
36struct lscreen_softc {
37 /* Initialized by config function */
38 struct spinlock ls_lock; // protects data and device regs
39 unsigned ls_width, ls_height; // screen size
40 unsigned ls_cx, ls_cy; // cursor position
41 char *ls_screen; // memory-mapped screen buffer
42
43 /* Initialized by lower-level attachment function */
44 void *ls_busdata; // bus we're on
45 uint32_t ls_buspos; // position on that bus
46
47 /* Initialized by higher-level attachment function */
48 void *ls_devdata; // data and functions for
49 void (*ls_start)(void *devdata); // upper device (perhaps
50 void (*ls_input)(void *devdata, int ch); // console)
51};
