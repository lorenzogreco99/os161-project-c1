47static
48int
49semfs_eachopen(struct vnode *vn, int openflags)
50{
51 struct semfs_vnode *semv = vn->vn_data;
52
53 if (semv->semv_semnum == SEMFS_ROOTDIR) {
54 if ((openflags & O_ACCMODE) != O_RDONLY) {
55 return EISDIR;
56 }
57 if (openflags & O_APPEND) {
58 return EISDIR;
59 }
60 }
61
62 return 0;
63}

135/*
136 * Wakeup helper. We only need to wake up if there are sleepers, which
137 * should only be the case if the old count is 0; and we only
138 * potentially need to wake more than one sleeper if the new count
139 * will be more than 1.
140 */
141static
142void
143semfs_wakeup(struct semfs_sem *sem, unsigned newcount)
144{
145 if (sem->sems_count > 0 || newcount == 0) {
146 return;
147 }
148 if (newcount == 1) {
149 cv_signal(sem->sems_cv, sem->sems_lock);
150 }
151 else {
152 cv_broadcast(sem->sems_cv, sem->sems_lock);
153 }
154}

156/*
157 * stat() for semaphore vnodes
158 */
159static
160int
161semfs_semstat(struct vnode *vn, struct stat *buf)
162{
163 struct semfs_vnode *semv = vn->vn_data;
164 struct semfs_sem *sem;
165
166 sem = semfs_getsem(semv);
167
168 bzero(buf, sizeof(*buf));
169
170 lock_acquire(sem->sems_lock);
171 buf->st_size = sem->sems_count;
172 buf->st_nlink = sem->sems_linked ? 1 : 0;
173 lock_release(sem->sems_lock);
174
175 buf->st_mode = S_IFREG | 0666;
176 buf->st_blocks = 0;
177 buf->st_dev = 0;
178 buf->st_ino = semv->semv_semnum;
179
180 return 0;
181}

183/*
184 * Read. This is P(); decrease the count by the amount read.
185 * Don't actually bother to transfer any data.
186 */
187static
188int
189semfs_read(struct vnode *vn, struct uio *uio)
190{
191 struct semfs_vnode *semv = vn->vn_data;
192 struct semfs_sem *sem;
193 size_t consume;
194
195 sem = semfs_getsem(semv);
196
197 lock_acquire(sem->sems_lock);
198 while (uio->uio_resid > 0) {
199 if (sem->sems_count > 0) {
200 consume = uio->uio_resid;
201 if (consume > sem->sems_count) {
202 consume = sem->sems_count;
203 }
204 DEBUG(DB_SEMFS, "semfs: sem%u: P, count %u -> %u\n",
205 semv->semv_semnum, sem->sems_count,
206 sem->sems_count - consume);
207 sem->sems_count -= consume;
208 /* don't bother advancing the uio data pointers */
209 uio->uio_offset += consume;
210 uio->uio_resid -= consume;
211 }
212 if (uio->uio_resid == 0) {
213 break;
214 }
215 if (sem->sems_count == 0) {
216 DEBUG(DB_SEMFS, "semfs: sem%u: blocking\n",
217 semv->semv_semnum);
218 cv_wait(sem->sems_cv, sem->sems_lock);
219 }
220 }
221 lock_release(sem->sems_lock);
222 return 0;
223}

225/*
226 * Write. This is V(); increase the count by the amount written.
227 * Don't actually bother to transfer any data.
228 */
229static
230int
231semfs_write(struct vnode *vn, struct uio *uio)
232{
233 struct semfs_vnode *semv = vn->vn_data;
234 struct semfs_sem *sem;
235 unsigned newcount;
236
237 sem = semfs_getsem(semv);
238
239 lock_acquire(sem->sems_lock);
240 while (uio->uio_resid > 0) {
241 newcount = sem->sems_count + uio->uio_resid;
242 if (newcount < sem->sems_count) {
243 /* overflow */
244 lock_release(sem->sems_lock);
245 return EFBIG;
246 }
247 DEBUG(DB_SEMFS, "semfs: sem%u: V, count %u -> %u\n",
248 semv->semv_semnum, sem->sems_count, newcount);
249 semfs_wakeup(sem, newcount);
250 sem->sems_count = newcount;
251 uio->uio_offset += uio->uio_resid;
252 uio->uio_resid = 0;
253 }
254 lock_release(sem->sems_lock);
255 return 0;
256}

258/*
259 * Truncate. Set the count to the specified value.
260 *
261 * This is slightly cheesy but it allows open(..., O_TRUNC) to reset a
262 * semaphore as one would expect. Also it allows creating semaphores
263 * and then initializing their counts to values other than zero.
264 */
265static
266int
267semfs_truncate(struct vnode *vn, off_t len)
268{
269 /* We should just use UINT_MAX but we don't have it in the kernel */
270 const unsigned max = (unsigned)-1;
271
272 struct semfs_vnode *semv = vn->vn_data;
273 struct semfs_sem *sem;
274 unsigned newcount;
275
276 if (len < 0) {
277 return EINVAL;
278 }
279 if (len > (off_t)max) {
280 return EFBIG;
281 }
282 newcount = len;
283
284 sem = semfs_getsem(semv);
285
286 lock_acquire(sem->sems_lock);
287 semfs_wakeup(sem, newcount);
288 sem->sems_count = newcount;
289 lock_release(sem->sems_lock);
290
291 return 0;
292}

297/*
298 * Directory read. Note that there's only one directory (the semfs
299 * root) that has all the semaphores in it.
300 */
301static
302int
303semfs_getdirentry(struct vnode *dirvn, struct uio *uio)
304{
305 struct semfs_vnode *dirsemv = dirvn->vn_data;
306 struct semfs *semfs = dirsemv->semv_semfs;
307 struct semfs_direntry *dent;
308 unsigned num, pos;
309 int result;
310
311 KASSERT(uio->uio_offset >= 0);
312 pos = uio->uio_offset;
313
314 lock_acquire(semfs->semfs_dirlock);
315
316 num = semfs_direntryarray_num(semfs->semfs_dents);
317 if (pos >= num) {
318 /* EOF */
319 result = 0;
320 }
321 else {
322 dent = semfs_direntryarray_get(semfs->semfs_dents, pos);
323 result = uiomove(dent->semd_name, strlen(dent->semd_name),
324 uio);
325 }
326
327 lock_release(semfs->semfs_dirlock);
328 return result;
329}

369/*
370 * Create a semaphore.
371 */
372static
373int
374semfs_creat(struct vnode *dirvn, const char *name, bool excl, mode_t mode,
375 struct vnode **resultvn)
376{
377 struct semfs_vnode *dirsemv = dirvn->vn_data;
378 struct semfs *semfs = dirsemv->semv_semfs;
379 struct semfs_direntry *dent;
380 struct semfs_sem *sem;
381 unsigned i, num, empty, semnum;
382 int result;
383
384 (void)mode;
385 if (!strcmp(name, ".") || !strcmp(name, "..")) {
386 return EEXIST;
387 }
388
389 lock_acquire(semfs->semfs_dirlock);
390 num = semfs_direntryarray_num(semfs->semfs_dents);
391 empty = num;
392 for (i=0; i<num; i++) {
393 dent = semfs_direntryarray_get(semfs->semfs_dents, i);
394 if (dent == NULL) {
395 if (empty == num) {
396 empty = i;
397 }
398 continue;
399 }
400 if (!strcmp(dent->semd_name, name)) {
401 /* found */
402 if (excl) {
403 lock_release(semfs->semfs_dirlock);
404 return EEXIST;
405 }
406 result = semfs_getvnode(semfs, dent->semd_semnum,
407 resultvn);
408 lock_release(semfs->semfs_dirlock);
409 return result;
410 }
411 }
412
413 /* create it */
414 sem = semfs_sem_create(name);
415 if (sem == NULL) {
416 result = ENOMEM;
417 goto fail_unlock;
418 }
419 lock_acquire(semfs->semfs_tablelock);
420 result = semfs_sem_insert(semfs, sem, &semnum);
421 lock_release(semfs->semfs_tablelock);
422 if (result) {
423 goto fail_uncreate;
424 }
425
426 dent = semfs_direntry_create(name, semnum);
427 if (dent == NULL) {
428 goto fail_uninsert;
429 }
430
431 if (empty < num) {
432 semfs_direntryarray_set(semfs->semfs_dents, empty, dent);
433 }
434 else {
435 result = semfs_direntryarray_add(semfs->semfs_dents, dent,
436 &empty);
437 if (result) {
438 goto fail_undent;
439 }
440 }
441
442 result = semfs_getvnode(semfs, semnum, resultvn);
443 if (result) {
444 goto fail_undir;
445 }
446
447 sem->sems_linked = true;
448 lock_release(semfs->semfs_dirlock);
449 return 0;
450
451 fail_undir:
452 semfs_direntryarray_set(semfs->semfs_dents, empty, NULL);
453 fail_undent:
454 semfs_direntry_destroy(dent);
455 fail_uninsert:
456 lock_acquire(semfs->semfs_tablelock);
457 semfs_semarray_set(semfs->semfs_sems, semnum, NULL);
458 lock_release(semfs->semfs_tablelock);
459 fail_uncreate:
460 semfs_sem_destroy(sem);
461 fail_unlock:
462 lock_release(semfs->semfs_dirlock);
463 return result;
464}

466/*
467 * Unlink a semaphore. As with other files, it may not actually
468 * go away if it's currently open.
469 */
470static
471int
472semfs_remove(struct vnode *dirvn, const char *name)
473{
474 struct semfs_vnode *dirsemv = dirvn->vn_data;
475 struct semfs *semfs = dirsemv->semv_semfs;
476 struct semfs_direntry *dent;
477 struct semfs_sem *sem;
478 unsigned i, num;
479 int result;
480
481 if (!strcmp(name, ".") || !strcmp(name, "..")) {
482 return EINVAL;
483 }
484
485 lock_acquire(semfs->semfs_dirlock);
486 num = semfs_direntryarray_num(semfs->semfs_dents);
487 for (i=0; i<num; i++) {
488 dent = semfs_direntryarray_get(semfs->semfs_dents, i);
489 if (dent == NULL) {
490 continue;
491 }
492 if (!strcmp(name, dent->semd_name)) {
493 /* found */
494 sem = semfs_getsembynum(semfs, dent->semd_semnum);
495 lock_acquire(sem->sems_lock);
496 KASSERT(sem->sems_linked);
497 sem->sems_linked = false;
498 if (sem->sems_hasvnode == false) {
499 lock_acquire(semfs->semfs_tablelock);
500 semfs_semarray_set(semfs->semfs_sems,
501 dent->semd_semnum, NULL);
502 lock_release(semfs->semfs_tablelock);
503 lock_release(sem->sems_lock);
504 semfs_sem_destroy(sem);
505 }
506 else {
507 lock_release(sem->sems_lock);
508 }
509 semfs_direntryarray_set(semfs->semfs_dents, i, NULL);
510 semfs_direntry_destroy(dent);
511 result = 0;
512 goto out;
513 }
514 }
515 result = ENOENT;
516 out:
517 lock_release(semfs->semfs_dirlock);
518 return result;
519}

521/*
522 * Lookup: get a semaphore by name.
523 */
524static
525int
526semfs_lookup(struct vnode *dirvn, char *path, struct vnode **resultvn)
527{
528 struct semfs_vnode *dirsemv = dirvn->vn_data;
529 struct semfs *semfs = dirsemv->semv_semfs;
530 struct semfs_direntry *dent;
531 unsigned i, num;
532 int result;
533
534 if (!strcmp(path, ".") || !strcmp(path, "..")) {
535 VOP_INCREF(dirvn);
536 *resultvn = dirvn;
537 return 0;
538 }
539
540 lock_acquire(semfs->semfs_dirlock);
541 num = semfs_direntryarray_num(semfs->semfs_dents);
542 for (i=0; i<num; i++) {
543 dent = semfs_direntryarray_get(semfs->semfs_dents, i);
544 if (dent == NULL) {
545 continue;
546 }
547 if (!strcmp(path, dent->semd_name)) {
548 result = semfs_getvnode(semfs, dent->semd_semnum,
549 resultvn);
550 lock_release(semfs->semfs_dirlock);
551 return result;
552 }
553 }
554 lock_release(semfs->semfs_dirlock);
555 return ENOENT;
556}

591/*
592 * Reclaim - drop a vnode that's no longer in use.
593 */
594static
595int
596semfs_reclaim(struct vnode *vn)
597{
598 struct semfs_vnode *semv = vn->vn_data;
599 struct semfs *semfs = semv->semv_semfs;
600 struct vnode *vn2;
601 struct semfs_sem *sem;
602 unsigned i, num;
603
604 lock_acquire(semfs->semfs_tablelock);
605
606 /* vnode refcount is protected by the vnode's ->vn_countlock */
607 spinlock_acquire(&vn->vn_countlock);
608 if (vn->vn_refcount > 1) {
609 /* consume the reference VOP_DECREF passed us */
610 vn->vn_refcount--;
611
612 spinlock_release(&vn->vn_countlock);
613 lock_release(semfs->semfs_tablelock);
614 return EBUSY;
615 }
616
617 spinlock_release(&vn->vn_countlock);
618
619 /* remove from the table */
620 num = vnodearray_num(semfs->semfs_vnodes);
621 for (i=0; i<num; i++) {
622 vn2 = vnodearray_get(semfs->semfs_vnodes, i);
623 if (vn2 == vn) {
624 vnodearray_remove(semfs->semfs_vnodes, i);
625 break;
626 }
627 }
628
629 if (semv->semv_semnum != SEMFS_ROOTDIR) {
630 sem = semfs_semarray_get(semfs->semfs_sems, semv->semv_semnum);
631 KASSERT(sem->sems_hasvnode);
632 sem->sems_hasvnode = false;
633 if (sem->sems_linked == false) {
634 semfs_semarray_set(semfs->semfs_sems,
635 semv->semv_semnum, NULL);
636 semfs_sem_destroy(sem);
637 }
638 }
639
640 /* done with the table */
641 lock_release(semfs->semfs_tablelock);
642
643 /* destroy it */
644 semfs_vnode_destroy(semv);
645 return 0;
646}

714/*
715 * Constructor for semfs vnodes.
716 */
717static
718struct semfs_vnode *
719semfs_vnode_create(struct semfs *semfs, unsigned semnum)
720{
721 const struct vnode_ops *optable;
722 struct semfs_vnode *semv;
723 int result;
724
725 if (semnum == SEMFS_ROOTDIR) {
726 optable = &semfs_dirops;
727 }
728 else {
729 optable = &semfs_semops;
730 }
731
732 semv = kmalloc(sizeof(*semv));
733 if (semv == NULL) {
734 return NULL;
735 }
736
737 semv->semv_semfs = semfs;
738 semv->semv_semnum = semnum;
739
740 result = vnode_init(&semv->semv_absvn, optable,
741 &semfs->semfs_absfs, semv);
742 /* vnode_init doesn't actually fail */
743 KASSERT(result == 0);
744
745 return semv;
746}

748/*
749 * Look up the vnode for a semaphore by number; if it doesn't exist,
750 * create it.
751 */
752int
753semfs_getvnode(struct semfs *semfs, unsigned semnum, struct vnode **ret)
754{
755 struct vnode *vn;
756 struct semfs_vnode *semv;
757 struct semfs_sem *sem;
758 unsigned i, num;
759 int result;
760
761 /* Lock the vnode table */
762 lock_acquire(semfs->semfs_tablelock);
763
764 /* Look for it */
765 num = vnodearray_num(semfs->semfs_vnodes);
766 for (i=0; i<num; i++) {
767 vn = vnodearray_get(semfs->semfs_vnodes, i);
768 semv = vn->vn_data;
769 if (semv->semv_semnum == semnum) {
770 VOP_INCREF(vn);
771 lock_release(semfs->semfs_tablelock);
