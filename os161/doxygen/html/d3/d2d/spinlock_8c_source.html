51spinlock_init(struct spinlock *splk)
52{
53 spinlock_data_set(&splk->splk_lock, 0);

55 HANGMAN_LOCKABLEINIT(&splk->splk_hangman, "spinlock");

68/*
69 * Get the lock.
70 *
71 * First disable interrupts (otherwise, if we get a timer interrupt we
72 * might come back to this lock and deadlock), then use a machine-level
73 * atomic operation to wait for the lock to be free.
74 */
75void
76spinlock_acquire(struct spinlock *splk)
77{
78 struct cpu *mycpu;
79
80 splraise(IPL_NONE, IPL_HIGH);
81
82 /* this must work before curcpu initialization */
83 if (CURCPU_EXISTS()) {
84 mycpu = curcpu->c_self;
85 if (splk->splk_holder == mycpu) {
86 panic("Deadlock on spinlock %p\n", splk);
87 }
88 mycpu->c_spinlocks++;
89
90 HANGMAN_WAIT(&curcpu->c_hangman, &splk->splk_hangman);
91 }
92 else {
93 mycpu = NULL;
94 }
95
96 while (1) {
97 /*
98 * Do test-test-and-set, that is, read first before
99 * doing test-and-set, to reduce bus contention.
100 *
101 * Test-and-set is a machine-level atomic operation
102 * that writes 1 into the lock word and returns the
103 * previous value. If that value was 0, the lock was
104 * previously unheld and we now own it. If it was 1,
105 * we don't.
106 */
107 if (spinlock_data_get(&splk->splk_lock) != 0) {
108 continue;
109 }
110 if (spinlock_data_testandset(&splk->splk_lock) != 0) {
111 continue;
112 }
113 break;
114 }
115
116 membar_store_any();
117 splk->splk_holder = mycpu;
118
119 if (CURCPU_EXISTS()) {
120 HANGMAN_ACQUIRE(&curcpu->c_hangman, &splk->splk_hangman);
121 }
122}

124/*
125 * Release the lock.
126 */
127void
128spinlock_release(struct spinlock *splk)
129{
130 /* this must work before curcpu initialization */
131 if (CURCPU_EXISTS()) {
132 KASSERT(splk->splk_holder == curcpu->c_self);
133 KASSERT(curcpu->c_spinlocks > 0);
134 curcpu->c_spinlocks--;
135 HANGMAN_RELEASE(&curcpu->c_hangman, &splk->splk_hangman);
136 }
137
138 splk->splk_holder = NULL;
139 membar_any_store();
140 spinlock_data_set(&splk->splk_lock, 0);
141 spllower(IPL_HIGH, IPL_NONE);
142}

144/*
145 * Check if the current cpu holds the lock.
146 */
147bool
148spinlock_do_i_hold(struct spinlock *splk)
149{
150 if (!CURCPU_EXISTS()) {
151 return true;
152 }
153
154 /* Assume we can read splk_holder atomically enough for this to work */
155 return (splk->splk_holder == curcpu->c_self);
156}
