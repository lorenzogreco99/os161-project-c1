40/* dimension of matrices (cannot be too large or will overflow stack) */

44/* number of iterations for sleepalot threads */
45#define SLEEPALOT_PRINTS 20 /* number of printouts */
46#define SLEEPALOT_ITERS 4 /* iterations per printout */

48#define WAKER_WAKES 100

52/* N distinct wait channels */
53#define NWAITCHANS 12
54static struct spinlock spinlocks[NWAITCHANS];
55static struct wchan *waitchans[NWAITCHANS];

57static volatile int wakerdone;

61static
62void
63setup(void)
64{
65 char tmp[16];
66 int i;
67
68 if (wakersem == NULL) {
69 wakersem = sem_create("wakersem", 1);
70 donesem = sem_create("donesem", 0);
71 for (i=0; i<NWAITCHANS; i++) {
72 spinlock_init(&spinlocks[i]);
73 snprintf(tmp, sizeof(tmp), "wc%d", i);
74 waitchans[i] = wchan_create(kstrdup(tmp));
75 }
76 }
77 wakerdone = 0;
78}

80static
81void
82sleepalot_thread(void *junk, unsigned long num)
83{
84 int i, j;
85
86 (void)junk;
87
88 for (i=0; i<SLEEPALOT_PRINTS; i++) {
89 for (j=0; j<SLEEPALOT_ITERS; j++) {
90 unsigned n;
91 struct spinlock *lk;
92 struct wchan *wc;
93
94 n = random() % NWAITCHANS;
95 lk = &spinlocks[n];
96 wc = waitchans[n];
97 spinlock_acquire(lk);
98 wchan_sleep(wc, lk);
99 spinlock_release(lk);
100 }
101 kprintf("[%lu]", num);
102 }
103 V(donesem);
104}

106static
107void
108waker_thread(void *junk1, unsigned long junk2)
109{
110 int i, done;
111
112 (void)junk1;
113 (void)junk2;
114
115 while (1) {
116 P(wakersem);
117 done = wakerdone;
118 V(wakersem);
119 if (done) {
120 break;
121 }
122
123 for (i=0; i<WAKER_WAKES; i++) {
124 unsigned n;
125 struct spinlock *lk;
126 struct wchan *wc;
127
128 n = random() % NWAITCHANS;
129 lk = &spinlocks[n];
130 wc = waitchans[n];
131 spinlock_acquire(lk);
132 wchan_wakeall(wc, lk);
133 spinlock_release(lk);
134
135 thread_yield();
136 }
137 }
138 V(donesem);
139}

161static
162void
163compute_thread(void *junk1, unsigned long num)
164{
165 struct matrix {
166 char m[DIM][DIM];
167 };
168 struct matrix *m1, *m2, *m3;
169 unsigned char tot;
170 int i, j, k, m;
171 uint32_t rand;
172
173 (void)junk1;
174
175 m1 = kmalloc(sizeof(struct matrix));
176 KASSERT(m1 != NULL);
177 m2 = kmalloc(sizeof(struct matrix));
178 KASSERT(m2 != NULL);
179 m3 = kmalloc(sizeof(struct matrix));
180 KASSERT(m3 != NULL);
181
182 for (m=0; m<COMPUTE_ITERS; m++) {
183
184 for (i=0; i<DIM; i++) {
185 for (j=0; j<DIM; j++) {
186 rand = random();
187 m1->m[i][j] = rand >> 16;
188 m2->m[i][j] = rand & 0xffff;
189 }
190 }
191
192 for (i=0; i<DIM; i++) {
193 for (j=0; j<DIM; j++) {
194 tot = 0;
195 for (k=0; k<DIM; k++) {
196 tot += m1->m[i][k] * m2->m[k][j];
197 }
198 m3->m[i][j] = tot;
199 }
200 }
201
202 tot = 0;
203 for (i=0; i<DIM; i++) {
204 tot += m3->m[i][i];
205 }
206
207 kprintf("{%lu: %u}", num, (unsigned) tot);
208 thread_yield();
209 }
210
211 kfree(m1);
212 kfree(m2);
213 kfree(m3);
214
215 V(donesem);
216}

234static
235void
236finish(int howmanytotal)
237{
238 int i;
239 for (i=0; i<howmanytotal; i++) {
240 P(donesem);
241 }
242 P(wakersem);
243 wakerdone = 1;
244 V(wakersem);
245 P(donesem);
