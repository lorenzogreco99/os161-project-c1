54/*
55 * Create a dummy struct thread that we can put on lists for testing.
56 */
57static
58struct thread *
59fakethread_create(const char *name)
60{
61 struct thread *t;
62
63 t = kmalloc(sizeof(*t));
64 if (t == NULL) {
65 panic("threadlisttest: Out of memory\n");
66 }
67 /* ignore most of the fields, zero everything for tidiness */
68 bzero(t, sizeof(*t));
69 t->t_name = kstrdup(name);
70 if (t->t_name == NULL) {
71 panic("threadlisttest: Out of memory\n");
72 }
73 t->t_stack = FAKE_MAGIC;
74 threadlistnode_init(&t->t_listnode, t);
75 return t;
76}

78/*
79 * Destroy a fake thread.
80 */
81static
82void
83fakethread_destroy(struct thread *t)
84{
85 KASSERT(t->t_stack == FAKE_MAGIC);
86 threadlistnode_cleanup(&t->t_listnode);
87 kfree(t->t_name);
88 kfree(t);
89}

94static
95void
96check_order(struct threadlist *tl, bool rev)
97{
98 const char string0[] = "...";
99 const char stringN[] = "~~~";
100
101 struct thread *t;
102 const char *first = rev ? stringN : string0;
103 const char *last = rev ? string0 : stringN;
104 const char *prev;
105 int cmp;
106
107 prev = first;
108 THREADLIST_FORALL(t, *tl) {
109 cmp = strcmp(prev, t->t_name);
110 KASSERT(rev ? (cmp > 0) : (cmp < 0));
111 prev = t->t_name;
112 }
113 cmp = strcmp(prev, last);
114 KASSERT(rev ? (cmp > 0) : (cmp < 0));
115}

277void
278threadlisttest_f(void)
279{
280 struct threadlist tl;
281 struct thread *t;
282 unsigned i;
283
284 threadlist_init(&tl);
285
286 for (i=0; i<NUMNAMES; i++) {
287 threadlist_addtail(&tl, fakethreads[i]);
288 }
289 KASSERT(tl.tl_count == NUMNAMES);
290
291 i=0;
292 THREADLIST_FORALL(t, tl) {
293 KASSERT(t == fakethreads[i]);
294 i++;
295 }
296 KASSERT(i == NUMNAMES);
297
298 i=0;
299 THREADLIST_FORALL_REV(t, tl) {
300 KASSERT(t == fakethreads[NUMNAMES - i - 1]);
301 i++;
302 }
303 KASSERT(i == NUMNAMES);
304
305 for (i=0; i<NUMNAMES; i++) {
306 t = threadlist_remhead(&tl);
307 KASSERT(t == fakethreads[i]);
308 }
309 KASSERT(tl.tl_count == 0);
310}
